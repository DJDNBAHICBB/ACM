#include <bits/stdc++.h>
using namespace std;
const int e = 1000000007;
const int maxn = 180000000;
int numtosq[maxn];
int numstat[maxn];
bool visit[maxn];
bool visit1[maxn];
void sieve(int n)
{
    for(int i=2;i<=n;i++)
    {
        if (visit[i])
            continue;
        long long k = i;
        while (k <= n)
        {
            numtosq[k] = i;
            visit[k] = true;
            k*=i;
        }
    }
}
signed main(){
    int n;
    cin >> n;
    sieve(n);
    long long  num = 3;
    while (num<=n)
    {
        for (int i = 2; i < num; ++i) {
            if ((num*1ll*i)>n)
            {
                break;
            }
            if (visit1[num*1ll*i])
                continue;
            if (numtosq[num] != numtosq[i]){
                visit1[num*1ll*i] = true;
                numstat[num]++;
                numstat[i]++;
            }
        }
        num++;
    }
    long long  sum = 1;
    for (int j = 2; j <= n; ++j) {
        if (numstat[j]%2==1)
            sum=(sum*j)%e;
    }
    cout <<  sum << endl;
}
