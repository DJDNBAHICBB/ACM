#include <bits/stdc++.h>

using namespace std;
const int e = 1000000007;
const int maxn = 180000000;
#define mem(x, i) memset(x,i,sizeof(x))
int prime[maxn];
bool visit[maxn];
bool vis[maxn];
int numsta[maxn];

void Prime(int n) {
    for (int i = 2; i <= n; i++) {
        if (!visit[i]) {
            prime[++prime[0]] = i;      //纪录素数， 这个prime[0] 相当于 cnt，用来计数
        }
        for (int j = 1; j <= prime[0] && i * prime[j] <= maxn; j++) {
            visit[i * prime[j]] = 1;
            if (i % prime[j] == 0) {
                break;
            }
        }
    }
}

signed main() {
    int n;
    cin >> n;
    Prime(n);
    for (int i = 1; i <= prime[0]; ++i) {
        if (prime[i] > n/2)
            break;
        for (int j = 1; j < i; ++j) {
            if (prime[i] * 1ll * prime[j] <= n) {
                if (!vis[prime[i] * prime[j]]) {
                    numsta[prime[i]]++;
                    numsta[prime[j]]++;
                    vis[prime[i] * prime[j]] = true;
                }
            } else
                break;
        }
    }
    long long sum = 1;
    for (int j = 2; j <= n; ++j) {
        if (numsta[j]>0)
            sum = (sum * j) % e;
    }
    cout << sum << endl;
}
